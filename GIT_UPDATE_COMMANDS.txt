# 1. 查看目前狀態
git status

# 2. 將修改加入暫存（只加入特定檔案可改成 git add path/to/file）
git add .

# 3. 提交（請修改訊息為具體變更說明）
git commit -m "更新：樣式與班級輸入功能"

# 4. 從遠端拉下最新（若遠端分支為 main，將 <branch> 換成 main 或 master）
git pull --rebase origin <branch>

# 5. 將本地提交推上遠端
git push origin <branch>

# 額外：若尚未設定遠端 repository
# git remote add origin https://github.com/username/repo.git
# git push -u origin <branch>

# 額外：若要查看遠端與分支
# git remote -v
# git branch -vv
